# VEX V5 Python Project
					import sys
					import vex
					from vex import *
					import motor_group
					import drivetrain
					import smartdrive

					#region config
					brain      = vex.Brain()
					motor_1    = vex.Motor(vex.Ports.PORT1, vex.GearSetting.RATIO18_1, False)
					motor_2    = vex.Motor(vex.Ports.PORT2, vex.GearSetting.RATIO18_1, False)
					motor_3    = vex.Motor(vex.Ports.PORT3, vex.GearSetting.RATIO18_1, True)
					motor_4    = vex.Motor(vex.Ports.PORT4, vex.GearSetting.RATIO18_1, True)
					inertial_5 = vex.Inertial(vex.Ports.PORT5)
					motor_6    = vex.Motor(vex.Ports.PORT6, vex.GearSetting.RATIO18_1, False)
					motor_7    = vex.Motor(vex.Ports.PORT7, vex.GearSetting.RATIO18_1, True)
					bumper_A   = vex.Bumper(brain.three_wire_port.a)
					sonar_C    = vex.Sonar(brain.three_wire_port.c)
					dt         = smartdrive.Smartdrive((motor_3, motor_4), (motor_1, motor_2), inertial_5, 319.1764, 292.1, vex.DistanceUnits.MM, 1)
					#endregion config


					#Brazo.set_velocity
					dt.set_drive_velocity(80,vex.PERCENT)
					dt.set_turn_velocity(80,vex.PERCENT)
					dt.set_stopping(vex.BrakeType.HOLD)

					motor_6.set_velocity(100, vex.PERCENT)
					motor_7.set_velocity(100, vex.PERCENT)


					dt.turn_for(vex.LEFT,90)
					dt.drive_for(vex.REVERSE,60,vex.DistanceUnits.CM)
					#depositar pelota
					dt.turn_for(vex.RIGHT,180)
					dt.drive_for(vex.REVERSE,80,vex.DistanceUnits.CM)
					#bajar brazo
					#recoger pelota
					#depositar pelota
					dt.turn_for(vex.LEFT,90)
					dt.drive_for(vex.REVERSE,170,vex.DistanceUnits.CM)
					dt.turn_for(vex.RIGHT,90)
					dt.drive_for(vex.REVERSE,30,vex.DistanceUnits.CM)
					dt.turn_for(vex.LEFT,90)
					dt.drive_for(vex.REVERSE,30,vex.DistanceUnits.CM)
					#REMOVER PELOTA

					dt.stop()
